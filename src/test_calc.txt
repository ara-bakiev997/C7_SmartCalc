
#include "s21_smart_calc.h"
#include <stdlib.h>
#include <ctype.h>
#include <math.h>
#include <stdio.h>
#include <string.h>

#suite Tsuite1
#tcase Tcase1

#test test_1_s21_smart_calc
int error = 1;
double res = 0;
error = launch("1+2-3", &res);
ck_assert_double_eq(res, 0);
ck_assert_int_eq(error, 1);

error = launch("sqrt(4)+9.235+2^3^sin(cos(10))-564-sqrt(25)+28.6*35mod99^2^3", &res);
ck_assert_double_eq(res, 444.59309217773205);
ck_assert_int_eq(error, 1);

error = launch("1+2+3+(3+5)*2+3/(1-4)^3*(3^((2+1)+1))/(((1)))+2", &res);
ck_assert_double_eq(res, 15);
ck_assert_int_eq(error, 1);

error = launch("-5+(-1+2)*4*cos(+2*7.5-2)+sin(cos(2*5))-sqrt(2^log(5-1))+ln(55/"
      "(2+7))+asin(0.5)+tan(50)", &res);
ck_assert_double_eq(res, -1.2844528706620064);
ck_assert_int_eq(error, 1);

error = launch("asin(2)", &res);
ck_assert_int_eq(error, 0);

error = launch("sqrt(-2)", &res);
ck_assert_int_eq(error, 0);

error = launch("10/0", &res);
ck_assert_int_eq(error, 0);

res = 0;
error = launch("+++7575", &res);
ck_assert_double_eq(res, 0);
ck_assert_int_eq(error, 0);

res = 0;
error = launch("+7..575", &res);
ck_assert_double_eq(res, 0);
ck_assert_int_eq(error, 0);

res = 0;
error = launch("+7575.", &res);
ck_assert_double_eq(res, 0);
ck_assert_int_eq(error, 0);
